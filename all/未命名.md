erDiagram

    t_admin {
        int id PK
        varchar username
        varchar password
        datetime create_time
    }

    t_staff {
        int id PK
        varchar username
        varchar password
        varchar real_name
        varchar phone
        varchar role
        datetime create_time
    }

    t_category {
        int id PK
        varchar name
        varchar description
        int product_num
    }

    t_supplier {
        int id PK
        varchar name
        varchar contact_person
        varchar phone
        varchar address
        varchar email
    }

    t_customer {
        int id PK
        varchar name
        varchar contact_person
        varchar phone
        varchar address
        varchar email
    }

    t_product {
        int id PK
        int category_id FK
        varchar name
        varchar specification
        varchar unit
        decimal purchase_price
        decimal sale_price
        int stock_num
        int safe_stock
    }

    t_purchase_order {
        bigint id PK
        int supplier_id FK
        int staff_id FK
        decimal total_amount
        datetime purchase_date
        varchar status
    }

    t_purchase_detail {
        bigint id PK
        bigint purchase_id FK
        int product_id FK
        int quantity
        decimal price
        decimal amount
    }

    t_sales_order {
        bigint id PK
        int customer_id FK
        int staff_id FK
        decimal total_amount
        datetime sales_date
        varchar status
    }

    t_sales_detail {
        bigint id PK
        bigint sales_id FK
        int product_id FK
        int quantity
        decimal price
        decimal amount
    }

    t_inventory_log {
        bigint id PK
        int product_id FK
        int staff_id FK
        varchar change_type
        int change_num
        int stock_before
        int stock_after
        datetime create_time
    }

    t_stock_check {
        bigint id PK
        int product_id FK
        int staff_id FK
        int system_stock
        int actual_stock
        int difference_num
        varchar check_type
        datetime check_time
    }

    t_category ||--o{ t_product : 分类

    t_supplier ||--o{ t_purchase_order : 供应

    t_purchase_order ||--o{ t_purchase_detail : 包含

    t_product ||--o{ t_purchase_detail : 采购商品

    t_customer ||--o{ t_sales_order : 下单

    t_staff ||--o{ t_sales_order : 创建

    t_sales_order ||--o{ t_sales_detail : 包含

    t_product ||--o{ t_sales_detail : 销售商品

    t_product ||--o{ t_inventory_log : 库存变动

    t_staff ||--o{ t_inventory_log : 操作

    t_product ||--o{ t_stock_check : 盘点

    t_staff ||--o{ t_stock_check : 执行
